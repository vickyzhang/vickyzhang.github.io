PHP date函数常用参数 

<php
echo date("j");//输出月份中的第几天 1~31天  返回的是string
echo date("t",strtotime("2013-02-12"));//给定月份的天数，（28-31）返回string 28
echo date('w'); //返回星期几（0为星期天，6为星期六）

//获取上月/当前月 的第一天/最后一天
echo date('the first day of last/this month');
echo date('the last day of last/this month');
?>
